#include<bits/stdc++.h>
using namespace std; 
const int maxn=1010;
int N,M,Price[maxn],G[maxn][maxn],best,Min_Price=INT_MAX,Stack[maxn][maxn],dp[maxn];
vector<int> A,ans,temp;
int mp[maxn],rmp[maxn],cnt=1;
void DFS(int idx,int total,int cnt,int price){
	if(total==0){
		if(cnt>best||(cnt==best&&price<Min_Price)){
			Min_Price=price;
			ans=temp;
			best=cnt;
		}
		return; 
	}
	for(int i=0;i<total;i++){
		if(cnt+total-i<best){
			return;
		}
		int u=Stack[idx][i];
		if(cnt+dp[u]<best){
			return;
		}
		int sz=0;
		for(int j=i+1;j<total;j++){
			int& v=Stack[idx][j];
			if(G[u][v]){
				Stack[idx+1][sz++]=v;
			}
		}
		temp.push_back(u);
		DFS(idx+1,sz,cnt+1,price+Price[u]);
		temp.pop_back();
	}
}
void Max_clique(){
	best=0;
	for(int i=N;i>0;i--){
		int sz=0;
		for(int j=i+1;j<=N;j++){
			if(G[i][j]){
				Stack[1][sz++]=j;
			}
		}
		temp.push_back(i);
		DFS(1,sz,1,Price[i]);
		temp.pop_back();
		dp[i]=best;
	}
}
int trans(int x){
	if(mp[x]==0){
		mp[x]=cnt++;	
		rmp[mp[x]]=x;
	}
	return mp[x];
}
int main() {
	std::ios::sync_with_stdio(false);
	fill(G[0],G[0]+maxn*maxn,1);
	cin>>M>>N;
	A.resize(N);
	for(int i=0;i<M;i++){
		int u,v;
		cin>>u>>v;
		u=trans(u);v=trans(v);
		G[u][v]=G[v][u]=0;
	}
	for(int i=0;i<N;i++){
		int v;
		cin>>A[i]>>v;
		A[i]=trans(A[i]);
		Price[A[i]]=v;
	}
	Max_clique();
	sort(ans.begin(),ans.end(),[](const int&a,const int&b){
		return rmp[a]<rmp[b];
	});
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++){
		cout<<setw(3)<<setfill('0')<<rmp[ans[i]]<<(i==ans.size()-1?"\n":" ");
	}
	cout<<Min_Price;
	return 0;
}

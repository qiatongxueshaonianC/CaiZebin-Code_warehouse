#include<bits/stdc++.h>
using namespace std; 
int main() {
	int N,D;
	double e,u;
	scanf("%d%lf%d",&N,&e,&D);
	pair<int,int> Ans(0,0);
	for(int i=0;i<N;i++){
		int k,maybe=0;
		scanf("%d",&k);
		for(int j=0;j<k;j++){
			scanf("%lf",&u);
			if(u<e)
				maybe++;
		}
		if(k>D&&maybe*2>k) Ans.second++;  //2020/1/27/15:06发现，这里一定要看清题目，是k>D!! 
		else if(maybe*2>k) Ans.first++;
	}
	printf("%.1lf%% %.1lf%%",100.0*Ans.first/N,100.0*Ans.second/N);
}

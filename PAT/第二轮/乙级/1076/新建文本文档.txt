#include<bits/stdc++.h>
using namespace std; 
int main() {
	int N;
	string s;
	scanf("%d%*c",&N);
	unordered_map<char,int> mp={{'A',1},{'B',2},{'C',3},{'D',4}};
	for(int i=0;i<N;i++){
		getline(cin,s);
		int k=s.find('T');
		printf("%d",mp[s[k-2]]);
	} 
	return 0;
}

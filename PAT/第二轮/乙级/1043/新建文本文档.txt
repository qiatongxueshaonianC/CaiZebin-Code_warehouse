		j=k+1;
		k=s.find_first_of("PATest",j);
	}
	int visit[6]={'P','A','T','e','s','t'};

		for(int i=0;i<6;i++){
			if(hash[visit[i]]>0){
				printf("%c",visit[i]);
				hash[visit[i]]--;
				flag=true;				
			}
		}
		if(flag==false)
			break;
	}
	return 0;
}

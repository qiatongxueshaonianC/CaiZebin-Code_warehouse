#include<bits/stdc++.h>
using namespace std; 
const int maxn=1010;
int main() {
	int n,group,id,score,A[maxn]={0},ans=0;
	memset(A,sizeof(A),0);
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d-%d %d",&group,&id,&score);
		A[group]+=score;
	}
	score=0;
	for(int i=0;i<maxn;i++){
		if(A[i]>=score){
			score=A[i];
			ans=i;
		}
	}
	printf("%d %d",ans,score);
	return 0;
}

#define lowbit(x) x&(-x0)
using namespace std;
typedef long long LL;
const int maxn=110;
LL w[maxn],N,M,Sum;
vector<vector<int> >G;
multiset<vector<int>,greater<vector<int> > >ans;
vector<int> temp;
void DFS(int id,LL sum){
	if(G[id].size()==0){
		temp.push_back(w[id]);
		sum+=w[id];
		if(sum==Sum){
			ans.insert(temp);
		}
		temp.pop_back();
		return;
	}
	temp.push_back(w[id]);
	sum+=w[id];
	if(sum>Sum)
		goto loop;
	for(auto i:G[id]){
		DFS(i,sum);
	}
	loop:
	temp.pop_back();
}
int main() {
	std::ios::sync_with_stdio(false);
	cin>>N>>M>>Sum;
	G.resize(N);
	for(int i=0;i<N;i++){
		cin>>w[i];
	}
	for(int i=0;i<M;i++){
		int id,k,u;
		cin>>id>>k;
		while(k--){
			cin>>u;
			G[id].push_back(u);
		}
	}
	DFS(0,0);
	for(auto I:ans){
		for(int i=0;i<I.size();i++){
			cout<<I[i]<<(i==I.size()-1?"\n":" ");
		}
	}
	return 0;
}

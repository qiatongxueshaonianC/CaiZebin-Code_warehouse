#include<bits/stdc++.h>
using namespace std; 
int Rank[10010];
bool Harsh[10010];
bool isPrime(int s){
	if(s<=1) return false;
	int sqr=(int)sqrt(1.0*s);
	for(int i=2;i<=sqr;i++)
		if(s%i==0)
			return false;  //非素数
	return true; 
}
int main() {
	int n,k,id;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&id);
		Rank[id]=i;
	}
	scanf("%d",&k);
	for(int i=0;i<k;i++){
		scanf("%d",&id);
		if(Harsh[id]==1)
			printf("%04d: Checked\n",id);
		else if(Rank[id]==1){
			printf("%04d: Mystery Award\n",id);
			Harsh[id]=1;
		}else if(Rank[id]==0){
			printf("%04d: Are you kidding?\n",id);
		}else if(isPrime(Rank[id])){
			printf("%04d: Minion\n",id);
			Harsh[id]=true;
		}else{
			printf("%04d: Chocolate\n",id);
			Harsh[id]=true;
		}	
	}
	return 0;
}

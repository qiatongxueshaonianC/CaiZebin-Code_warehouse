#include<bits/stdc++.h>
using namespace std; 
vector<pair<int,int>> edge; 
int main() {
	int n,m,k,u,v;
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++){
		scanf("%d%d",&u,&v);
		edge.push_back({u,v});
	}
	scanf("%d",&k);
	while(k--){
		unordered_set<int> st;
		int harsh[n];
		for(int i=0;i<n;i++){
			scanf("%d",&harsh[i]);
			st.insert(harsh[i]);
		}
		bool flag=true;
		for(auto i:edge){
			if(harsh[i.first]==harsh[i.second]){
				flag=false;
				break;
			}
		}
		if(flag==false)
			puts("No");
		else
			printf("%d-coloring\n",st.size());
	}
	return 0;
}

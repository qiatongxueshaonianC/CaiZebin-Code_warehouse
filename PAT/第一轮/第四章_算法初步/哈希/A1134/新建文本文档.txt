#include<bits/stdc++.h>
using namespace std; 
multiset<pair<int,int>> edge;
int main() {
	int n,m,k,u,v,ans;
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++){
		scanf("%d%d",&u,&v);
		edge.insert({u,v});
	} 
	scanf("%d",&k);
	while(k--){
		bool harsh[n+1]={false},flag=true;
		int num=0;
		scanf("%d",&m);
		for(int i=0;i<m;i++){
			scanf("%d",&v);
			harsh[v]=true;
		}
		for(auto i:edge){
			if(harsh[i.first]==false&&harsh[i.second]==false){
				flag=false;
				break;
			}
		}
		printf("%s\n",flag?"Yes":"No");
	}
	return 0;
}

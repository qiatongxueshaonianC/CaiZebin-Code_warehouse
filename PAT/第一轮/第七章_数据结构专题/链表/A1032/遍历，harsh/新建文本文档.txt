#include<bits/stdc++.h>
using namespace std; 
struct Node{
	char data;
	int id,next,flag;
	Node():flag(-1){}
}L[100005];
int main() {
	int l1,l2,n,id,next;
	char c;
	scanf("%d%d%d",&l1,&l2,&n);
	for(int i=0;i<n;i++){
		scanf("%d ",&id);
		L[id].id=id;
		scanf("%c %d",&L[id].data,&L[id].next);
	}
	while(l1!=-1){
		L[l1].flag=1;
		l1=L[l1].next;
	}
	while(l2!=-1){
		if(L[l2].flag==1) break;
		l2=L[l2].next;
	}
	if(l2==-1) puts("-1");
	else printf("%05d\n",L[l2].id);
	return 0;
}

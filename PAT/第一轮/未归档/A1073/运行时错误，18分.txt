#include<bits/stdc++.h>
using namespace std; 
string deal(string s){
	int k=s.find('E');
	char c=s[k+1];
	int num=stoi(s.substr(k+2));
	s.erase(k);
//	puts(s.c_str());
	s.erase(2,1);
	if(c=='+')
		s.insert(s.end(),num-(k-2-1),'0');
	else {
		s.insert(1,num,'0');
		s.insert(2,".");		
	}
	if(s[0]=='+')
		s.erase(s.begin());
	return s;
}
int main() {
	string s;
	getline(cin,s);
	s=deal(s);
	puts(s.c_str());
	return 0;
}

#include<bits/stdc++.h>
using namespace std; 
bool cmp(int a,int b){
	return a>b;
}
int main() {
	int n,i=0;
	scanf("%d",&n);
	int A[n];
	for(int i=0;i<n;i++)
		scanf("%d",&A[i]);
	sort(A,A+n,cmp);
	while(i<n&&A[i]>i+1) 
		i++;
	printf("%d\n",i);
	return 0;
}

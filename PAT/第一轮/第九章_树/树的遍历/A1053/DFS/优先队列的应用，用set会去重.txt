#include<bits/stdc++.h>
using namespace std;
const int maxn=110;
struct Node{
	int weight;
	vector<int> child;
}Tree[maxn];
int n,m,Sqr;
vector<int> temp;
priority_queue<vector<int> >ans;
//priority_queue<vector<int>,vector<vector<int> >,greater<vector<int> > >ans;  //ÈôÊÇ°´ÕÕ×ÖµäÐò 
void DFS(int s,int val){
	if(Tree[s].child.size()==0&&val==Sqr){
		temp.push_back(Tree[s].weight);
		ans.push(temp);
		temp.pop_back();
		return;
	}
	temp.push_back(Tree[s].weight);
	for(auto i:Tree[s].child){
		if(val+Tree[i].weight>Sqr) continue;
		DFS(i,val+Tree[i].weight);
	}
	temp.pop_back();
}
int main() {
	scanf("%d%d%d",&n,&m,&Sqr);
	int id,num,v;
	for(int i=0;i<n;i++){
		scanf("%d",&v);
		Tree[i].weight=v;
	}
	for(int i=0;i<m;i++){
		scanf("%d%d",&id,&num);
		for(int i=0;i<num;i++){
			scanf("%d",&v);
			Tree[id].child.push_back(v);
		}
	}
	DFS(0,Tree[0].weight);
	while(!ans.empty()){
		vector<int> now=ans.top();ans.pop();
		for(int i=0;i<now.size();i++){
			printf("%d",now[i]);
			if(i<now.size()-1) printf(" ");
		}
		puts("");
	}
	return 0;
}

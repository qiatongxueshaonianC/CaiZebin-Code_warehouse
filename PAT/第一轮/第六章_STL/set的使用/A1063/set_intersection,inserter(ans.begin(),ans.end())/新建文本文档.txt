#include<bits/stdc++.h>
using namespace std; 
int main() {
	int n,m,u,v;
	scanf("%d",&n);
	vector<set<int>> A(n+1);
	for(int i=1;i<=n;i++){
		scanf("%d",&u);
		while(u--){
			scanf("%d",&v);
			A[i].insert(v);
		}
	}
	scanf("%d",&m);
	while(m--){
		scanf("%d%d",&u,&v);
		set<int> ans;
		set_intersection(A[u].begin(),A[u].end(),A[v].begin(),A[v].end(),inserter(ans,ans.begin()));
		printf("%.1lf%\n",100.0*ans.size()/(A[u].size()+A[v].size()-ans.size()));
	}
	return 0;
}

#include<bits/stdc++.h>
using namespace std;
unordered_map<string,set<int>>mp;
int main() {
	int query,n;
	string name;
	scanf("%d%d",&query,&n);
	while(n--){
		int course_id,num;
		scanf("%d%d",&course_id,&num);
		while(num--){
			cin>>name;
			if(mp.find(name)==mp.end()){
				set<int> temp;
				temp.insert(course_id);
				mp[name]=temp;
			}else{
				mp[name].insert(course_id);
			}
		}
	}
	for(int i=0;i<query;i++){
		cin>>name;
		printf("%s %d",name.c_str(),mp[name].size());
		for(auto i:mp[name])
			printf(" %d",i);
		puts("");
	}
	return 0;
}

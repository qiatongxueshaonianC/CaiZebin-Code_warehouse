#include<bits/stdc++.h>
using namespace std;
unordered_map<int,set<string>> mp;
int main() {
	int n,k;
	string name;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++){
		int num,v;
		cin>>name>>num;
		while(num--){
			scanf("%d",&v);
			if(mp.find(v)==mp.end()){
				set<string> temp;
				temp.insert(name);
				mp[v]=temp;
			}else
				mp[v].insert(name);
		}
	}
	for(int i=1;i<=k;i++){
		printf("%d %d\n",i,mp[i].size());
		for(auto j:mp[i])
			printf("%s\n",j.c_str());
	}
	return 0;
}
